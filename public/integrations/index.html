Use SnapAPI everywhere your code runs.

📦 GitHub Actions — Auto-Generate Social Cards on PR

Generate social preview images for your site when you open a pull request.

name: Generate Social Preview
on: [pull_request]
jobs:
  preview:
    runs-on: ubuntu-latest
    steps:
      - run: |
        curl -H "X-Api-Key: ${{ secrets.SNAPAPI_KEY }}" \
        "https://snapapi.loca.lt/api/screenshot?url=${{ args.url }}" \
        -o preview.png

⚡ Vercel — Dynamic OG Images

Generate OG images at build time with a serverless function.

// api/og.ts — Vercel Edge Function
export default async (req) => {
  const url = new URL(req.url).searchParams.get('url');
  const res = await fetch(`https://snapapi.loca.lt/api/screenshot?url=${url}&key=${process.env.SNAPAPI_KEY}`);
  return new Response(res.body, { headers: { 'Content-Type': 'image/png' } });
};
